SELECT
REPLACE(REPLACE(REPLACE(REPLACE(REPLACE(type_leave, '0', 'ลาบวช-ทำหมัน') , '1', 'ลาคลอด') , '2', 'ลาป่วย'), '3', 'ลากิจ'), '4', 'พักร้อน') AS type_leave, 
DATE_FORMAT(nstart_day,'%m') AS months,
COUNT(nstart_day) AS count_leave 
FROM
`leaves`
GROUP BY type_leave,DATE_FORMAT(nstart_day,'%Y-%m')

SELECT
`leaves`.id_per,
libs.surname,
`leaves`.type_leave,
Count(`leaves`.nstart_day) AS count_person
FROM
libs
INNER JOIN `leaves` ON `leaves`.id_per = libs.id
GROUP BY leaves.id_per,type_leave

`leaves`
